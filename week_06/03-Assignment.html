\[\begin{split} \begin{eqnarray} b & = & \frac{n\sum_{i=1}^n x_iy_i - \sum_{i=1}^n x_i \sum_{i=1}^ny_i} {n\sum_{i=1}^n (x_i)^2 - (\sum_{i=1}^n x_i)^2} \\ a & = & \overline{y} - b\overline{x} \end{eqnarray} \end{split}\]
